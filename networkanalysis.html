      #centered at an average location
      center = [sum(lat for lat, lon in coordinates) / len(coordinates), 
                sum(lon for lat, lon in coordinates) / len(coordinates)]
      mymap = Map(center=center, zoom=10, basemap=basemaps.OpenStreetMap.Mapnik)
      # Add markers for each point in the coordinates list
      for lat, lon in coordinates:
          marker = Marker(location=(lat, lon), draggable=False)
          mymap.add_layer(marker)
      mymap
